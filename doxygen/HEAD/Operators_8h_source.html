17 template<typename Derived1, typename Derived2, typename Derived3>
18 inline void colwiseCrossEq(const Eigen::MatrixBase<Derived1> & m1,
19  const Eigen::MatrixBase<Derived2> & m2,
20  Eigen::MatrixBase<Derived3> const & result)
21 {
22  Eigen::MatrixBase<Derived3> & result_nc = const_cast<Eigen::MatrixBase<Derived3> &>(result);
23  result_nc.row(0) = (m1.row(1) * m2.coeff(2) - m1.row(2) * m2.coeff(1));
24  result_nc.row(1) = (m1.row(2) * m2.coeff(0) - m1.row(0) * m2.coeff(2));
25  result_nc.row(2) = (m1.row(0) * m2.coeff(1) - m1.row(1) * m2.coeff(0));
26 }

51 inline void colwiseLeftMultEq(const Eigen::MatrixBase<Derived1> & m1,

53  Eigen::MatrixBase<Derived3> const & result)
54 {

394 template<typename T>
395 inline RBInertia<T> PTransform<T>::dualMul(const RBInertia<T> & rbI) const
396 {
397  using namespace Eigen;
398  const Matrix3<T> & E = rotation();
399  const Vector3<T> & r = translation();
400  Matrix3<T> I;
401  I.template triangularView<Lower>() =
402  E
403  * (rbI.inertia() + vector3ToCrossMatrix(r) * vector3ToCrossMatrix<T>(rbI.momentum())
404  + vector3ToCrossMatrix<T>(rbI.momentum() - rbI.mass() * r) * vector3ToCrossMatrix<T>(r))
405  * E.transpose();
406  return RBInertia<T>(rbI.mass(), E * (rbI.momentum() - rbI.mass() * r), I);

423 template<typename T>
424 inline ABInertia<T> PTransform<T>::dualMul(const ABInertia<T> & rbI) const
425 {
426  using namespace Eigen;
427  const Matrix3<T> & E = rotation();
428  const Vector3<T> & r = translation();
429 
430  Matrix3<T> massM = rbI.massMatrix();
431  Matrix3<T> rCross = vector3ToCrossMatrix(r);
432  Matrix3<T> tmpI = rbI.gInertia() - rCross * massM;
433 
434  Matrix3<T> M, I;
435  M.template triangularView<Lower>() = E * massM * E.transpose();
436  I.template triangularView<Lower>() =
437  E * (rbI.inertia() - rCross * rbI.gInertia().transpose() + (tmpI * rCross)) * E.transpose();
438 
439  return ABInertia<T>(M, E * (tmpI)*E.transpose(), I);
440 }

442 template<typename T>
443 inline ABInertia<T> PTransform<T>::transMul(const ABInertia<T> & rbI) const
444 {
445  using namespace Eigen;
446  const Matrix3<T> & E = rotation();
447  const Vector3<T> & r = translation();
448 
449  Matrix3<T> Mp(E.transpose() * rbI.massMatrix() * E);
450  Matrix3<T> Hp(E.transpose() * rbI.gInertia() * E);
451  Matrix3<T> rCross(vector3ToCrossMatrix(r));
452 
453  Matrix3<T> M, I;
454  M.template triangularView<Lower>() = Mp;
455  I.template triangularView<Lower>() =
456  (E.transpose() * rbI.inertia() * E + rCross * Hp.transpose() - (Hp + rCross * Mp) * rCross);
457  return ABInertia<T>(M, Hp + rCross * Mp, I);
458 }
