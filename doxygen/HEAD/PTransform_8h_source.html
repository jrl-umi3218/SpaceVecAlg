292 template<typename T>
293 inline Vector3<T> rotationVelocity(const Matrix3<T> & E_a_b)
294 {
295  Vector3<T> w;
296  T acosV = (E_a_b(0, 0) + E_a_b(1, 1) + E_a_b(2, 2) - 1.) * 0.5;
297  T theta = std::acos(std::min(std::max(acosV, -1.), 1.));
298 
299  w = Vector3<T>(-E_a_b(2, 1) + E_a_b(1, 2), -E_a_b(0, 2) + E_a_b(2, 0), -E_a_b(1, 0) + E_a_b(0, 1));
300  w *= sinc_inv(theta) * 0.5;
301 
302  return w;
303 }

321 {
322  Eigen::Quaternion<T> qfrom(from.rotation());
323  Eigen::Quaternion<T> qto(to.rotation());
